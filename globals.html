Const createCngnApiClient

  • createCngnApiClient(config: CngnApiClientConfig): { createVirtualAccount: createVirtualAccount; formatApiError: formatApiError; formatApiResponse: formatApiResponse; get: get; getBalance: getBalance; getBanks: getBanks; getTransactionHistory: getTransactionHistory; post: post; redeemAsset: redeemAsset; swapAsset: swapAsset; updateExternalAccounts: updateExternalAccounts; verifyWithdrawal: verifyWithdrawal; withdraw: withdraw }
    • Defined in lib/cngn.ts:58

    Creates a configured API client for interacting with CNGN endpoints.

    All POST requests automatically encrypt their payload using AES-256-CBC via the imported crypto utilities. Additionally, responses are decrypted using the provided encryptionKey (used as the Ed25519 private key) and then wrapped in a standardized FormattedResponse.

    Example (ES module - using async/await)

     
    import { createCngnApiClient } from './cngnApiClient';

(async () => {
  const client = createCngnApiClient({
    apiVersion: 'v1',
    apiKey: 'YOUR_API_KEY',
    encryptionKey: 'YOUR_ENCRYPTION_KEY', // required for both encryption and decryption
    errorHandler: (error) => console.error("API error:", error)
  });
  try {
    const balance = await client.getBalance();
    console.log(balance);
  } catch (err) {
    console.error(err);
  }
})();

Const createCryptoUtils

  • createCryptoUtils(encryptionKey: string): { decryptData: decryptData; encryptData: encryptData }
    • Defined in lib/aes.ts:51

    Creates a utility object for performing AES-256-CBC encryption and decryption.

    This function returns an object with two methods:

    • encryptData: Encrypts a plain text string.
    • decryptData: Decrypts an EncryptedData object back to its original plain text.

    Example (ES module)

     
    import { createCryptoUtils } from './cryptoUtils';

const cryptoUtils = createCryptoUtils("myEncryptionKey");

// Encrypt plain text data
const encrypted = cryptoUtils.encryptData("Hello, world!");
console.log(encrypted);
// Expected output:
// { iv: "base64-encoded-iv", content: "base64-encoded-ciphertext" }

// Decrypt the previously encrypted data
const decrypted = cryptoUtils.decryptData(encrypted);
console.log(decrypted);
// Expected output:
// "Hello, world!"
